Start -> PROGRAM <DECLARATIONS> BEGIN <STATEMENTS> END;

VAR_ASSIGN -> VAR := <MATH_EXPR>;
IF_STMT -> IF <REL_EXPR> THEN <STATEMENTS> <ELSE_STMT> END IF;
DO_STMT -> DO <STATEMENTS> UNTIL <REL_EXPR>;
FOR_STMT -> FOR VAR := INT_LIT , INT_LIT [INT_LIT] <STATEMENTS> END FOR;
WHILE_STMT -> WHILE <REL_EXPR> BEGIN [<STATEMENTS>] END WHILE;
BIN_EXPR -> OPERAND BIN_OP OPERAND
UN_EXPR -> UN_OP OPERAND
OPERAND -> INT_LIT | FLOAT_LIT | VAR
REL_EXPRE -> OPERAND REL_OP OPERAND
PRINT_STMT -> PRINT(<ARGUMENT>);

STMT -> IF <EXPR> THEN <STATEMENTS> <TAIL_STATEMENT> END IF;
TAIL_STATEMENT -> ELSE <STATEMENTS> | E
STATEMENTS -> STMT ; TAIL_STATEMENT | STMT | E

<PROGRAM> -> BEGIN <STMTS> END;
<STMTS> -> SIMPLESTMT; <STMTS>
<STMTS> -> BEGIN <STMTS> END; <STMTS>
<STMTS> -> LAMBDA
